void InputMatrix(int ***A,int*m,int*n)
{
	int i,j ;
	printf("请输入m，n\n");
	scanf("%d%d",m,n);
	*A=(int**)malloc(sizeof(int*)**m);
	for(i=0;i<*m;i++)
	{
		*(*A+i)=(int*)malloc(sizeof(int)**n);
	}
	printf("请输入矩阵A(%d*%d)\n",*m,*n);
	for(i=0;i<*m;i++)
		for(j=0;j<*n;j++)
		{
			scanf("%d",*(*A+i)+j);
		}
}
//注意：如果传入三级指针，内存分配分别为*A=（）malloc（）和循环中*（*A+i）=（）malloc（）
             如果将二级指针作为返回值，则写A=（）malloc（）和循环中*（A+i）=（）malloc（）
